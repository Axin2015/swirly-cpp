INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/include")
INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/src")

ADD_DEFINITIONS(-DDBR_HIDE_API)

FILE(GLOB ash_SOURCES "*.c")

ADD_LIBRARY(ash_static STATIC ${ash_SOURCES})
SET_TARGET_PROPERTIES(ash_static PROPERTIES OUTPUT_NAME ash)
TARGET_LINK_LIBRARIES(ash_static)

ADD_LIBRARY(ash_pic STATIC ${ash_SOURCES})
SET_TARGET_PROPERTIES(ash_pic PROPERTIES COMPILE_FLAGS "${CMAKE_SHARED_LIBRARY_C_FLAGS}")
TARGET_LINK_LIBRARIES(ash_pic)

INSTALL(TARGETS ash_static DESTINATION lib)
