INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/include")
INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/src")

FILE(GLOB ash_SOURCES "*.c")

ADD_LIBRARY(ash_shared SHARED ${ash_SOURCES})
SET_TARGET_PROPERTIES(ash_shared PROPERTIES OUTPUT_NAME dbr_ash)
TARGET_LINK_LIBRARIES(ash_shared m)

ADD_LIBRARY(ash_static STATIC ${ash_SOURCES})
SET_TARGET_PROPERTIES(ash_static PROPERTIES OUTPUT_NAME dbr_ash)
TARGET_LINK_LIBRARIES(ash_static m)

ADD_LIBRARY(ash_pic STATIC ${ash_SOURCES})
SET_TARGET_PROPERTIES(ash_pic PROPERTIES OUTPUT_NAME dbr_ash_pic)
SET_TARGET_PROPERTIES(ash_pic PROPERTIES COMPILE_FLAGS "${CMAKE_SHARED_LIBRARY_C_FLAGS}")
TARGET_LINK_LIBRARIES(ash_pic m)

INSTALL(TARGETS ash_shared DESTINATION lib)
INSTALL(TARGETS ash_static DESTINATION lib)
