INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/include")
INCLUDE_DIRECTORIES("${PROJECT_SOURCE_DIR}/src")

FILE(GLOB fir_SOURCES "*.c")

ADD_LIBRARY(fir_shared SHARED ${fir_SOURCES})
SET_TARGET_PROPERTIES(fir_shared PROPERTIES OUTPUT_NAME dbr_fir)
TARGET_LINK_LIBRARIES(fir_shared ash_shared)

ADD_LIBRARY(fir_static STATIC ${fir_SOURCES})
SET_TARGET_PROPERTIES(fir_static PROPERTIES OUTPUT_NAME dbr_fir)
TARGET_LINK_LIBRARIES(fir_static ash_static)

ADD_LIBRARY(fir_pic STATIC ${fir_SOURCES})
SET_TARGET_PROPERTIES(fir_pic PROPERTIES OUTPUT_NAME dbr_fir_pic)
SET_TARGET_PROPERTIES(fir_pic PROPERTIES COMPILE_FLAGS "${CMAKE_SHARED_LIBRARY_C_FLAGS}")
TARGET_LINK_LIBRARIES(fir_pic ash_pic)

INSTALL(TARGETS fir_shared DESTINATION lib)
INSTALL(TARGETS fir_static DESTINATION lib)
